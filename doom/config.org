(use-package! org-excalidraw)

(after! org-excalidraw
  (defun org-excalidraw-to-svg-thumbnail (file)
    "Export excalidraw FILE to svg thumbnail."
    (let* ((path (expand-file-name file))
           (svg-path (org-excalidraw-svg-thumbnail-path path)))
      (unless (string-suffix-p ".excalidraw" path)
        (error "Excalidraw diagrams must end with .excalidraw extension."))
      (cond
       ((executable-find "kroki")
        (shell-command
         (format "kroki convert \"%s\" --type excalidraw --format svg --out-file \"%s\""
                 path svg-path)))
       ((executable-find "excalidraw_export")
        (shell-command
         (format "excalidraw_export --ember_fonts --rename_fonts \"%s\"" path)))
       (t
        (error "Can't convert %s to svg." file))))))

** GTD
#+BEGIN_SRC emacs-lisp
(after! org
  ;; ── Agenda files ─────────────────────────────────────────────────────────
  (setq org-agenda-files
        (list (concat org-directory "/todo.org")
              (concat org-directory "/tickler.org")))

  ;; ── Keywords ─────────────────────────────────────────────────────────────
  (setq org-todo-keywords
        '((sequence "TODO(t)" "NEXT(n)" "WAITING(w@/!)" "SOMEDAY(s)"
                    "|" "DONE(d!)" "CANCELLED(c@)")))

  (setq org-todo-keyword-faces
        '(("TODO"      . (:foreground "#7c7c75" :weight bold))
          ("NEXT"      . (:foreground "#0098dd" :weight bold))
          ("WAITING"   . (:foreground "#9f7efe" :weight bold))
          ("SOMEDAY"   . (:foreground "#e6c547" :weight bold))
          ("DONE"      . (:foreground "#50a14f" :weight bold))
          ("CANCELLED" . (:foreground "#ff6c6b" :weight bold))))

  ;; ── Tags ─────────────────────────────────────────────────────────────────
  (setq org-tag-alist
        '((:startgroup)
          ("@home"     . ?h)
          ("@office"   . ?o)
          ("@computer" . ?c)
          ("@phone"    . ?p)
          ("@errands"  . ?e)
          (:endgroup)
          ("PROJECT"   . ?P)
          ("WAITING"   . ?W)
          ("SOMEDAY"   . ?S)
          ("sprint"    . ?s)))

  ;; ── Capture templates ────────────────────────────────────────────────────
  (setq org-capture-templates
        `(("i" "Inbox (quick capture)" entry
           (file ,(concat org-directory "/todo.org"))
           "* TODO %?\n:PROPERTIES:\n:CREATED: %U\n:END:\n"
           :empty-lines-after 1)

          ("n" "Next Action" entry
           (file+headline ,(concat org-directory "/todo.org") "Standalone Actions")
           "* NEXT %?\t\t\t:%^{Context|@computer|@phone|@home|@office|@errands}:\n:PROPERTIES:\n:CREATED: %U\n:END:\n"
           :empty-lines-after 1)

          ("p" "New Project" entry
           (file+headline ,(concat org-directory "/todo.org") "Projects")
           "* PROJECT %?\n:PROPERTIES:\n:CREATED: %U\n:END:\n"
           :empty-lines-after 1)

          ("s" "Someday / Maybe" entry
           (file+headline ,(concat org-directory "/someday.org") "Ideas & Projects")
           "* SOMEDAY %?\n:PROPERTIES:\n:CREATED: %U\n:END:\n"
           :empty-lines-after 1)

          ("t" "Tickler (reminder)" entry
           (file+headline ,(concat org-directory "/tickler.org") "Reminders")
           "* TODO %?\nSCHEDULED: %^t\n:PROPERTIES:\n:CREATED: %U\n:END:\n"
           :empty-lines-after 1)

          ("r" "Reference" entry
           (file+headline ,(concat org-directory "/reference.org") "Tech Notes")
           "* %?\n:PROPERTIES:\n:CREATED: %U\n:END:\n"
           :empty-lines-after 1)))

  ;; ── Refile ───────────────────────────────────────────────────────────────
  (setq org-refile-targets
        `((,(concat org-directory "/todo.org")      :maxlevel . 3)
          (,(concat org-directory "/someday.org")   :level    . 1)
          (,(concat org-directory "/tickler.org")   :maxlevel . 2)
          (,(concat org-directory "/reference.org") :level    . 1)))
  (setq org-refile-use-outline-path        t)
  (setq org-outline-path-complete-in-steps nil)

  ;; ── Logging ──────────────────────────────────────────────────────────────
  (setq org-log-done       'time)
  (setq org-log-into-drawer t)

  ;; ── Agenda display ───────────────────────────────────────────────────────
  (setq org-agenda-skip-scheduled-if-done t)
  (setq org-agenda-skip-deadline-if-done  t)
  (setq org-return-follows-link           t)

  ;; ── Custom agenda views ──────────────────────────────────────────────────
  (setq org-agenda-custom-commands
        '(("g" "GTD Dashboard"
           ((agenda ""
                    ((org-agenda-span 'day)
                     (org-agenda-overriding-header "Today")))
            (todo "NEXT"
                  ((org-agenda-overriding-header "Next Actions")
                   (org-super-agenda-groups
                    '((:name "@computer" :tag "@computer")
                      (:name "@phone"    :tag "@phone")
                      (:name "@home"     :tag "@home")
                      (:name "@office"   :tag "@office")
                      (:name "@errands"  :tag "@errands")
                      (:name "Other"     :anything t)))))
            (todo "WAITING"
                  ((org-agenda-overriding-header "Waiting For")))
            (tags "PROJECT"
                  ((org-agenda-overriding-header "Projects")
                   (org-tags-match-list-sublevels nil)))))

)))
